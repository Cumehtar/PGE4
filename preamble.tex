\usepackage[T1]{fontenc}
\usepackage{titlesec}
\usepackage{geometry}
\usepackage{longtable, booktabs}
\usepackage[hidelinks]{hyperref}

% --- COMPATIBLE ARCHAIC FONTS ---
% Palatino is a classic humanist serif (standard in all TeX distros)
\usepackage{palatino} 
% pzc is the PostScript Zapf Chancery font (standard cursive)
\newcommand{\handwriting}{\fontfamily{pzc}\selectfont}
% pbk is Bookman, which is heavy and "old-press" looking
\newcommand{\headerfont}{\fontfamily{pbk}\selectfont}

% 1. Geometry & Margin setup
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Sidenote definition (Using Zapf Chancery for handwritten feel)
\renewcommand{\footnote}[1]{%
  \marginpar{%
    \raggedright
    \handwriting
    \large % Chancery runs small, so large looks like normal handwriting
    #1
  }%
}

% 3. Header formatting & Page breaking
\newcommand{\sectionbreak}{\clearpage}

% # Header: Centered on its own page
\titleformat{\section}[display]
  {\headerfont\huge\bfseries\centering}{}{0pt}{\vspace*{\fill}}[\vfill]

% ## Header: Large with a rule
\titleformat{\subsection}
  {\headerfont\Large\bfseries}{}{0pt}{}[\vspace{10pt}\hrule]

% ### Header: Normal with a rule
\titleformat{\subsubsection}
  {\headerfont\large\bfseries}{}{0pt}{}[\vspace{5pt}\hrule]
