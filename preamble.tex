\usepackage{titlesec}
\usepackage{geometry}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}
\usepackage{longtable}
\usepackage{booktabs}

% 1. Geometry - Maintaining the wide margin for sidenotes
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Sidenotes (Handwritten style)
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 3. The "New Page" Rule
% This forces a clear page BEFORE every #, ##, and ### header
\newcommand{\sectionbreak}{\clearpage}

% 4. Header Styling
% Level 1 (#): Vertically Centered Splash Page
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}{}{0pt}{\vspace*{\fill}}[\vfill]

% Level 2 (##): Large Header with Rule
\titleformat{\subsection}
  {\normalfont\Large\bfseries}{}{0pt}{}[\vspace{10pt}\hrule]

% Level 3 (###): Standard Header with Rule
\titleformat{\subsubsection}
  {\normalfont\large\bfseries}{}{0pt}{}[\vspace{5pt}\hrule]

% 5. Settings
\usepackage[hidelinks]{hyperref}
