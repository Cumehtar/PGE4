% 3. Robust Table Patch
% This version handles the [] and {} arguments Pandoc throws at it
\makeatletter
\let\oldlongtable\longtable
\let\endoldlongtable\endlongtable

\renewenvironment{longtable}[2][]{% 
  \begin{center}
  % If the second argument (alignment) is empty, default to 'l'
  \if\relax\detokenize{#2}\relax
    \begin{tabular}{l}
  \else
    \begin{tabular}{#2}
  \fi
}{%
  \end{tabular}
  \end{center}
}
\makeatother

% 4. Column Switch Logic
\newbool{columnOpen}
\boolfalse{columnOpen}

\newcommand{\forceClose}{%
  \ifbool{columnOpen}{\end{multicols}\boolfalse{columnOpen}}{}%
}

% 5. Header Redefinitions (The Wrapper Method)
\let\oldsection\section
\renewcommand{\section}[1]{\forceClose\oldsection{#1}}

\let\oldsubsection\subsection
\renewcommand{\subsection}[1]{%
  \forceClose\oldsubsection{#1}%
  \begin{multicols}{3}\booltrue{columnOpen}%
}

\let\oldsubsubsection\subsubsection
\renewcommand{\subsubsection}[1]{%
  \forceClose\oldsubsubsection{#1}%
  \begin{multicols}{2}\booltrue{columnOpen}%
}
