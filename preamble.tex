\usepackage{titlesec}
\usepackage{geometry}
\usepackage[framemethod=TikZ]{mdframed}
\usepackage{xcolor}
\usepackage[T1]{fontenc}

% 1. Geometry - Wide right margin for sidenotes and overhang
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Perfectly Centered Chapters (#)
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace*{\fill}} 
[\vfill\clearpage]

% 3. Handwritten-style Sidenotes (Monospace Italic)
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 4. The Overhang Environment (###)
% This environment pushes 1.5in into the side margin
\newenvironment{OverhangBox}[1]{%
  \begin{list}{}{
    \leftmargin=0pt \rightmargin=-1.5in
    \listparindent=\parindent \itemindent=\parindent \parsep=\parskip
  }\item[]
  \begin{mdframed}[
    linecolor=black, linewidth=1.5pt, backgroundcolor=white,
    innertopmargin=10pt, innerbottommargin=10pt,
    innerleftmargin=10pt, innerrightmargin=10pt,
    skipabove=20pt, skipbelow=20pt
  ]
  \textbf{\large #1}\par\medskip
}{%
  \endmdframed\end{list}%
}

% Ensure links and table of contents are strictly Black & White
\usepackage[hidelinks]{hyperref}
