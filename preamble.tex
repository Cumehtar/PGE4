\usepackage{titlesec}
\usepackage{geometry}
\usepackage{multicol}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}

% 1. Geometry
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Footnotes as Sidenotes
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 3. Safe Column Switching Logic
\newbool{columnOpen}
\boolfalse{columnOpen}

\newcommand{\safeCloseColumns}{%
  \ifbool{columnOpen}{%
    \end{multicols}%
    \boolfalse{columnOpen}%
  }{}%
}

% Patch sectioning commands to close columns BEFORE starting a new header
\pretocmd{\section}{\safeCloseColumns}{}{}
\pretocmd{\subsection}{\safeCloseColumns}{}{}
\pretocmd{\subsubsection}{\safeCloseColumns}{}{}

% 4. Header Style Definitions (Clean)

% Level 1 (#): Centered Splash
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace*{\fill}} 
[\vfill\clearpage]

% Level 2 (##): 3 Columns
\titleformat{\subsection}
  {\normalfont\Large\bfseries}
  {}
  {0pt}
  {\begin{multicols}{3}\booltrue{columnOpen}}
  [\vspace{10pt}\hrule\vspace{10pt}]

% Level 3 (###): 2 Columns
\titleformat{\subsubsection}
  {\normalfont\large\bfseries}
  {}
  {0pt}
  {\begin{multicols}{2}\booltrue{columnOpen}}
  [\vspace{5pt}\hrule\vspace{5pt}]

% 5. Settings & Cleanup
\setlength{\columnsep}{20pt}
\setlength{\columnseprule}{0.4pt}
\usepackage[hidelinks]{hyperref}

% Ensure everything closes at the end
\AtEndDocument{\safeCloseColumns}
