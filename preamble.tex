% --- REQUIRED PACKAGES ---
\usepackage{fontspec}
\usepackage{titlesec}
\usepackage{geometry}
\usepackage{longtable, booktabs}
\usepackage{fancyhdr}
\usepackage[hidelinks]{hyperref}

% --- CLEAN CLEAR PAGES ---
% This redefines \cleardoublepage to ensure the empty page is actually empty
\makeatletter
\def\cleardoublepage{\clearpage\if@twoside \ifodd\c@page\else
  \hbox{}
  \thispagestyle{empty}
  \newpage
  \if@twocolumn\hbox{}\newpage\fi\fi\fi}
\makeatother

%  [Font and Geometry configurations ]
\pagestyle{fancy}
\fancyhf{} 

% --- CENTERING LOGIC ---
\fancyhfoffset[LE,RO]{0.5\dimexpr\marginparwidth+\marginparsep\relax}

% Header/Footer configuration
\fancyhead[RE,LO]{\headerfont\footnotesize\leftmark}
\fancyhead[LE,RO]{\headerfont\footnotesize\thepage}
\fancyfoot[C]{}

% --- SECTION/CHAPTER FORMATTING ---

% Ensure section-title pages remain empty
\assignpagestyle{\section}{empty}
\renewcommand{\headrulewidth}{0pt}

% Redefining section breaks to ensure they don't carry headers
\newcommand{\sectionbreak}{\cleardoublepage}
\newcommand{\subsectionbreak}{\clearpage}
\newcommand{\subsubsectionbreak}{\clearpage}

% This centers the header vertically on its own page
\titleformat{\section}[display]
  {\headerfont\huge\centering}
  {}
  {0pt}
  {\vspace*{\fill}}
  [\vfill\cleardoublepage]

\titleformat{\subsection}
  {\headerfont\large\centering}{}{10pt}{}

\titleformat{\subsubsection}
  {\headerfont\large}{}{5pt}{}
