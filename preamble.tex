\usepackage{titlesec}
\usepackage{geometry}
\usepackage{multicol}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}

% 1. Geometry - Standard margins for sidenotes
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Footnotes as Sidenotes (Handwritten style)
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 3. Header Styles & Automatic Column Switching

% Level 1 (#): Centered Chapter, starts 1-column mode
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\begin{multicols}{1}\vspace*{\fill}} 
[\vfill\clearpage\end{multicols}]

% Level 2 (##): 3-Column Layout (Data/Index Style)
\titleformat{\subsection}
  {\end{multicols}\clearpage\begin{multicols}{3}\normalfont\Large\bfseries}
  {}
  {0pt}
  {}
  [\vspace{10pt}\hrule\vspace{10pt}]

% Level 3 (###): 2-Column Layout (Standard Content)
\titleformat{\subsubsection}
  {\end{multicols}\begin{multicols}{2}\normalfont\large\bfseries}
  {}
  {0pt}
  {}
  [\vspace{5pt}\hrule\vspace{5pt}]

% 4. Global Settings
\setlength{\columnsep}{20pt} % Space between columns
\setlength{\columnseprule}{0.4pt} % Thin vertical line between columns
\usepackage[hidelinks]{hyperref}
