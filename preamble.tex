\usepackage{titlesec}
\usepackage{geometry}
\usepackage[most]{tcolorbox}

% 1. Wide margins for sidenotes
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Centered Chapters (#) in the middle of the page
% \section corresponds to Level 1 (#) headers
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace{6cm}} 
[\vfill\clearpage]

% 3. Redefine footnotes as sidenotes
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\footnotesize #1}%
}

% 4. Turn ### Subsubsections into Framed Boxes
% This style creates a box where the title is inside the frame
\titleformat{\subsubsection}
  {\normalfont}
  {}
  {0pt}
  {\begin{tcolorbox}[
    colback=white,
    colframe=black,
    sharp corners,
    boxrule=0.8pt,
    fonttitle=\bfseries,
    title=#1,
    left=5pt, right=5pt, top=5pt, bottom=5pt
  ]\ignorespaces}

% This "closes" the box after the subsubsection content ends
\newcommand{\subsubboxend}{\end{tcolorbox}}

% Note: Since titlesec doesn't easily "wrap" following text, 
% the text following ### will be inside the box until the next header.
