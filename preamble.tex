\usepackage[T1]{fontenc}
\usepackage{titlesec}
\usepackage{geometry}
\usepackage{longtable, booktabs}
\usepackage[hidelinks]{hyperref}

\usepackage[utf8]{inputenc}

% --- ARCHAIC & HANDWRITTEN FONTS ---
\usepackage{ebgaramond} % Main body (Renaissance Serif)
\usepackage{auncial}    % Headers (Medieval Uncial)
\usepackage{calligra}   % Sidebar (Handwritten Calligraphy)
% -----------------------------------


% 1. Geometry & Margin setup
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Sidenote definition: Now using Calligra for a handwritten look
\renewcommand{\footnote}[1]{%
  \marginpar{%
    \raggedright
    \calligra % Switch to calligraphic font
    \small    % Calligra runs small, so 'small' or 'normalsize' works best
    #1
  }%
}


% 3. Header formatting & Page breaking
\newcommand{\sectionbreak}{\clearpage}

% # Header: Centered on its own page using Uncial script
\titleformat{\section}[display]
  {\unclfamily\huge\centering}{}{0pt}{\vspace*{\fill}}[\vfill]

% ## Header: Large with a rule
\titleformat{\subsection}
  {\unclfamily\Large}{}{0pt}{}[\vspace{10pt}\hrule]

% ### Header: Normal with a rule
\titleformat{\subsubsection}
  {\unclfamily\large}{}{0pt}{}[\vspace{5pt}\hrule]
