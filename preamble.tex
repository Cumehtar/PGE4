\usepackage{titlesec}
\usepackage{geometry}

% 1. Only number Level 1 headers (# Chapters)
% 0 = parts, 1 = chapters/sections, 2 = subsections
\setcounter{secnumdepth}{1}

% 2. Wide margins for sidenotes
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 3. Centered Chapters (#) in the middle of the page
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace{6cm}} 
[\vfill\clearpage]

% 4. Redefine footnotes as sidenotes
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\footnotesize #1}%
}
