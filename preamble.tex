\usepackage{titlesec}
\usepackage{geometry}
\usepackage[framemethod=TikZ]{mdframed}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}
\usepackage{changepage} % Required for margin overhanging

% 1. Geometry Setup (Wide right margin for notes and overhang)
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Perfectly Centered Chapters (#)
% Uses flexible "springs" (\fill) to find the vertical center of the text block
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace*{\fill}} 
[\vfill\clearpage]

% 3. "Handwritten" Sidenotes
% Using Italic Typewriter font for a scratchy, B&W-safe organic look
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 4. Overhanging Subsubsection Boxes (###)
% Pushes 1.5 inches into the 2.5 inch margin
\mdfdefinestyle{overhangStyle}{
    linecolor=black,
    linewidth=1.5pt,
    backgroundcolor=white,
    innertopmargin=10pt,
    innerbottommargin=10pt,
    innerleftmargin=10pt,
    innerrightmargin=10pt,
    skipabove=20pt,
    skipbelow=20pt,
    rightmargin=-1.5in
}

\titleformat{\subsubsection}[block]
  {\normalfont\bfseries\large}
  {}
  {0pt}
  {\begin{adjustwidth}{0pt}{-1.5in}
   \begin{mdframed}[style=overhangStyle]}

% 5. Pure LaTeX Safety Hooks
% This closes the box/margin adjustment whenever a higher-level header starts
\newcommand{\closeoverhang}{%
  \unskip\strut\endmdframed\end{adjustwidth}%
}

\pretocmd{\section}{\closeoverhang}{}{}
\pretocmd{\subsection}{\closeoverhang}{}{}

% Ensure links are strictly Black & White
\usepackage[hidelinks]{hyperref}
