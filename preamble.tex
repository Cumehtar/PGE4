\usepackage{titlesec}
\usepackage{geometry}
\usepackage[framemethod=TikZ]{mdframed}
\usepackage{xcolor}
% Standard handwriting font
\usepackage{calligra}
\usepackage[T1]{fontenc}

% 1. Wide margins for sidenotes
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Centered Chapters (#)
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace{6cm}} 
[\vfill\clearpage]

% 3. Handwritten Sidenotes (Black & White)
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\calligra\small #1}%
}

% 4. Black and White Framed Subsubsections (###)
\mdfdefinestyle{sidebarStyle}{
    linecolor=black,
    linewidth=1.5pt,
    backgroundcolor=white,
    innertopmargin=10pt,
    innerbottommargin=10pt,
    innerleftmargin=10pt,
    innerrightmargin=10pt,
    skipabove=15pt,
    skipbelow=15pt
    % 'needlines' removed to fix kvsetkeys error
}

\titleformat{\subsubsection}[block]
  {\normalfont\bfseries\large}
  {}
  {0pt}
  {\begin{mdframed}[style=sidebarStyle]}
  [\end{mdframed}]
