\usepackage{titlesec}
\usepackage{geometry}
\usepackage{multicol}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}

% 1. Geometry
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Footnotes as Sidenotes
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 3. Safe Column Logic
\newbool{columnOpen}
\boolfalse{columnOpen}

\newcommand{\safeCloseColumns}{%
  \ifbool{columnOpen}{\end{multicols}\boolfalse{columnOpen}}{}%
}

\newcommand{\safeOpenColumns}[1]{%
  \safeCloseColumns
  \begin{multicols}{#1}
  \booltrue{columnOpen}
}

% 4. Header Definitions

% Level 1 (#): Splash Page (1 Column)
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\safeCloseColumns\vspace*{\fill}} 
[\vfill\clearpage]

% Level 2 (##): 3 Columns
\titleformat{\subsection}
  {\safeOpenColumns{3}\normalfont\Large\bfseries}
  {}
  {0pt}
  {}
  [\vspace{10pt}\hrule\vspace{10pt}]

% Level 3 (###): 2 Columns
\titleformat{\subsubsection}
  {\safeOpenColumns{2}\normalfont\large\bfseries}
  {}
  {0pt}
  {}
  [\vspace{5pt}\hrule\vspace{5pt}]

% 5. Global Settings
\setlength{\columnsep}{20pt}
\setlength{\columnseprule}{0.4pt}
\usepackage[hidelinks]{hyperref}

% Final Cleanup: Automatically close columns at the very end of the document
\AtEndDocument{\safeCloseColumns}
