\usepackage{titlesec}
\usepackage{geometry}
\usepackage{multicol}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}

% 1. Geometry
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Footnotes as Sidenotes
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 3. Switch Logic
\newbool{columnOpen}
\boolfalse{columnOpen}

\newcommand{\safeClose}{%
  \ifbool{columnOpen}{\end{multicols}\boolfalse{columnOpen}}{}%
}

% 4. Header Styles (Clean Formatting Only)
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}{}{0pt}{\vspace*{\fill}}[\vfill\clearpage]

\titleformat{\subsection}
  {\normalfont\Large\bfseries}{}{0pt}{}[\vspace{10pt}\hrule\vspace{10pt}]

\titleformat{\subsubsection}
  {\normalfont\large\bfseries}{}{0pt}{}[\vspace{5pt}\hrule\vspace{5pt}]

% 5. The "Bouncer" Hooks (The real fix)
% Before any header, close columns.
\pretocmd{\section}{\safeClose}{}{}
\pretocmd{\subsection}{\safeClose}{}{}
\pretocmd{\subsubsection}{\safeClose}{}{}

% After Sub/Subsub headers, open columns.
\apptocmd{\subsection}{\begin{multicols}{3}\booltrue{columnOpen}}{}{}
\apptocmd{\subsubsection}{\begin{multicols}{2}\booltrue{columnOpen}}{}{}

% 6. Settings
\setlength{\columnsep}{20pt}
\setlength{\columnseprule}{0.4pt}
\usepackage[hidelinks]{hyperref}

\AtEndDocument{\safeClose}
