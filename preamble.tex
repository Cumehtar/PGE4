\usepackage{titlesec}
\usepackage{geometry}
\usepackage[framemethod=TikZ]{mdframed}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage{etoolbox}

% 1. Geometry (Standard wide margin)
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Perfectly Centered Chapters
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}
  {}
  {0pt}
  {\vspace*{\fill}} 
[\vfill\clearpage]

% 3. "Handwritten" Sidenotes (Safe Mono)
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 4. Native Overhanging Box Logic
% We define a command to start a "wide" environment using a list
\newcommand{\startoverhang}{%
  \begin{list}{}{
    \leftmargin=0pt
    \rightmargin=-1.5in
    \listparindent=\parindent
    \itemindent=\parindent
    \parsep=\parskip
  }\item[]
  \begin{mdframed}[
    linecolor=black,
    linewidth=1.5pt,
    backgroundcolor=white,
    innertopmargin=10pt,
    innerbottommargin=10pt,
    innerleftmargin=10pt,
    innerrightmargin=10pt,
    skipabove=20pt,
    skipbelow=20pt
  ]
}

\newcommand{\closeoverhang}{%
  \endmdframed\end{list}%
}

% 5. Map Subsubsection (###) to the native overhang
\titleformat{\subsubsection}[block]
  {\normalfont\bfseries\large}
  {}
  {0pt}
  {\startoverhang}

% 6. Safety Hooks to close boxes automatically
\pretocmd{\section}{\closeoverhang}{}{}
\pretocmd{\subsection}{\closeoverhang}{}{}

% Ensure B&W links
\usepackage[hidelinks]{hyperref}
