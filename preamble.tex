\usepackage{fontspec}
\usepackage{titlesec}
\usepackage{geometry}
\usepackage{longtable, booktabs}
\usepackage[hidelinks]{hyperref}

% --- FONT CONFIGURATION FROM LOCAL DIRECTORY ---

% 1. Main Body Font (e.g., Garamond)
\setmainfont{Asul-Regular.ttf.ttf}[
    Path = ./fonts/,
    BoldFont = Asul-Bold.ttf
]

% 2. Header Font (e.g., An Archaic Uncial or Blackletter)
\newfontfamily\headerfont{Kaya_Medium.otf}[
    Path = ./fonts/
]

% 3. Sidebar/Handwritten Font
\newfontfamily\sidebarfont{khand.ttf}[
    Path = ./fonts/
]

% --- LAYOUT & MARGINS ---

\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 4. Sidenote definition using the local sidebar font
\renewcommand{\footnote}[1]{%
  \marginpar{%
    \raggedright
    \sidebarfont
    \small 
    #1
  }%
}

% 5. Header formatting
\newcommand{\sectionbreak}{\clearpage}

% # Header: Centered
\titleformat{\section}[display]
  {\headerfont\huge\centering}{}{0pt}{\vspace*{\fill}}[\vfill]

% ## Header: Large with a rule
\titleformat{\subsection}
  {\headerfont\Large}{}{0pt}{}[\vspace{10pt}\hrule]

% ### Header: Normal with a rule
\titleformat{\subsubsection}
  {\headerfont\large}{}{0pt}{}[\vspace{5pt}\hrule]
