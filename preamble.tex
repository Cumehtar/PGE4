% 1. Geometry
\geometry{
  left=1in, 
  right=2.5in, 
  marginparwidth=2in, 
  marginparsep=0.2in,
  footskip=0.5in
}

% 2. Sidenotes
\renewcommand{\footnote}[1]{%
  \marginpar{\raggedright\ttfamily\itshape\scriptsize #1}%
}

% 3. The "Table Patch": Make longtable work in columns
% We redefine longtable to act like a standard center/tabular block
\makeatletter
\let\oldlongtable\longtable
\let\endoldlongtable\endlongtable
\renewenvironment{longtable}[1][]{%
  \begin{center}
  \begin{tabular}{#1}
}{%
  \end{tabular}
  \end{center}
}
\makeatother

% 4. Manual Column Logic
\newbool{columnOpen}
\boolfalse{columnOpen}

\newcommand{\forceClose}{%
  \ifbool{columnOpen}{\end{multicols}\boolfalse{columnOpen}}{}%
}

% 5. Redefine Headers
\let\oldsection\section
\renewcommand{\section}[1]{\forceClose\oldsection{#1}}

\let\oldsubsection\subsection
\renewcommand{\subsection}[1]{%
  \forceClose\oldsubsection{#1}%
  \begin{multicols}{3}\booltrue{columnOpen}%
}

\let\oldsubsubsection\subsubsection
\renewcommand{\subsubsection}[1]{%
  \forceClose\oldsubsubsection{#1}%
  \begin{multicols}{2}\booltrue{columnOpen}%
}

% 6. Styling
\titleformat{\section}[display]
  {\normalfont\huge\bfseries\centering}{}{0pt}{\vspace*{\fill}}[\vfill\clearpage]

\titleformat{\subsection}
  {\normalfont\Large\bfseries}{}{0pt}{}[\vspace{10pt}\hrule\vspace{10pt}]

\titleformat{\subsubsection}
  {\normalfont\large\bfseries}{}{0pt}{}[\vspace{5pt}\hrule\vspace{5pt}]
