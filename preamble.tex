\usepackage{fontspec}
\usepackage{titlesec}
\usepackage{geometry}
\usepackage{longtable, booktabs}
\usepackage[hidelinks]{hyperref}

% --- FONT CONFIGURATION ---

\setmainfont{Asul-Regular.ttf.ttf}[
    Path = ./fonts/,
    BoldFont = Asul-Bold.ttf
]

\newfontfamily\headerfont{Kaya_Medium.otf}[
    Path = ./fonts/
]

\newfontfamily\sidebarfont{khand.ttf}[
    Path = ./fonts/
]

% --- LAYOUT & MARGINS ---
% 'twoside' enables mirrored margins (inner/outer)
% Increased 'outer' and 'marginparwidth' for bigger sidebars
\geometry{
  twoside,
  inner=0.75in, 
  outer=3.5in, 
  marginparwidth=3in, 
  marginparsep=0.25in,
  footskip=0.5in
}

% --- SIDENOTE CONFIGURATION ---
\renewcommand{\footnote}[1]{%
  \marginpar{%
    \raggedright
    \sidebarfont
    \footnotesize % Changed from \small to \footnotesize for smaller text
    #1
  }%
}

% --- SECTION/CHAPTER FORMATTING ---

\newcommand{\sectionbreak}{\clearpage}

% This centers the header vertically on its own page
% and \clearpage at the end ensures no text follows it on the same page
\titleformat{\section}[display]
  {\headerfont\huge\centering}
  {}
  {0pt}
  {\vspace*{\fill}}
  [\vfill\clearpage]

\titleformat{\subsection}
  {\headerfont\Large}{}{0pt}{}[\vspace{10pt}\hrule]

\titleformat{\subsubsection}
  {\headerfont\large}{}{0pt}{}[\vspace{5pt}\hrule]
