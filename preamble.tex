% --- REQUIRED PACKAGES ---
\usepackage{fontspec}
\usepackage{titlesec}
\usepackage{geometry}
\usepackage{longtable, booktabs}
\usepackage{fancyhdr}
\usepackage[hidelinks]{hyperref}

% --- CLEAN CLEAR PAGES ---
% Redefines \cleardoublepage to ensure that pages inserted to maintain 
% right-hand starts are completely empty (no headers/footers).
\makeatletter
\def\cleardoublepage{\clearpage\if@twoside \ifodd\c@page\else
  \hbox{}
  \thispagestyle{empty}
  \newpage
  \if@twocolumn\hbox{}\newpage\fi\fi\fi}
\makeatother

% --- FONT CONFIGURATION ---
\setmainfont{Flanker Griffo.otf}[
    Path = ./fonts/,
    BoldFont = Flanker Griffo Bold.otf,
    ItalicFont = Flanker Griffo Italic.otf,
    BoldItalicFont = Flanker Griffo Italic Bold.otf
]

\newfontfamily\headerfont{Kaya_Medium.otf}[
    Path = ./fonts/
]

\newfontfamily\sidebarfont{hoffm___.ttf}[
    Path = ./fonts/
]

% --- LAYOUT & MARGINS ---
% 'twoside' enables mirrored margins (inner/outer)
\geometry{
  a4paper,
  twoside,
  inner=0.75in, 
  outer=3in, 
  marginparwidth=2.5in, 
  marginparsep=0.25in,
  footskip=0.5in
}

% --- COLUMN SETUP ---
\setlength{\columnsep}{0.25in}
\newenvironment{twocol}{\begin{multicols}{2}}{\end{multicols}}

% --- PAGE STYLE & HEADERS ---
\pagestyle{fancy}
\fancyhf{} % Clear defaults

% CENTERING LOGIC: Shift headers/footers to center on the physical paper
\fancyhfoffset[LE,RO]{0.5\dimexpr\marginparwidth+\marginparsep\relax}

% Header: Section title on the inner side, Page number on the outer side
\fancyhead[RE,LO]{\headerfont\footnotesize\leftmark}
\fancyhead[LE,RO]{\headerfont\footnotesize\thepage}

% Footer: Empty
\fancyfoot[C]{}

% Remove the header line for a cleaner look
\renewcommand{\headrulewidth}{0pt}

% --- SIDENOTE CONFIGURATION ---
% Redirects standard footnotes to the sidenotes package logic
\renewcommand{\footnote}[1]{%
  \marginpar{%
    \raggedright
    \sidebarfont
    \footnotesize
    \ignorespaces #1
  }%
}

% --- SECTION/CHAPTER FORMATTING ---

% Ensure the page containing the section title is empty
\assignpagestyle{\section}{empty}

% Section breaks to ensure sections start on odd pages
\newcommand{\sectionbreak}{\cleardoublepage}
\newcommand{\subsectionbreak}{\clearpage}
\newcommand{\subsubsectionbreak}{\clearpage}

% Vertical centering for Section Titles on their own page
% The \cleardoublepage in the suffix ensures content starts on the next odd page
\titleformat{\section}[display]
  {\headerfont\huge\centering}
  {}
  {0pt}
  {\thispagestyle{empty}\vspace*{\fill}}
  [\vfill\cleardoublepage]

\titleformat{\subsection}
  {\headerfont\large\centering}{}{10pt}{}

\titleformat{\subsubsection}
  {\headerfont\large}{}{5pt}{}
